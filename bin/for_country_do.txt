FROM_MONTH="2016-12"
FOR_YEARS="2016 2017 2018"
COUNTRIES="Afghanistan Syrien Irak Iran Nigeria Eritrea Pakistan Somalia Russische Tuerkei Ungeklaert Sierra Aethiopien Jemen Libyen Mali"
# create per country csv
for COUNTRY in ${COUNTRIES}; do ./per_country.sh -c ${COUNTRY} -y "$FOR_YEARS"; done

for COUNTRY in ${COUNTRIES}; do ./cut_country.sh -f ../cooked/${COUNTRY}.csv -d ${FROM_MONTH}; done
# create csv for decisions excl. sonstige
for COUNTRY in ${COUNTRIES}; do ./cut_country_nixsonst.sh -f ../cooked/${COUNTRY}.csv -d ${FROM_MONTH}; done
# commit all
for COUNTRY in ${COUNTRIES}; do git commit ../cooked/${COUNTRY}.csv -m "adding ${MONTH} ${YEAR}"; done
for COUNTRY in ${COUNTRIES}; do git commit ../cooked/${COUNTRY}_cut.csv -m "adding ${MONTH} ${YEAR}"; done
for COUNTRY in ${COUNTRIES}; do git commit ../cooked/${COUNTRY}_cut_nixsonst.csv -m "adding ${MONTH} ${YEAR}"; done
